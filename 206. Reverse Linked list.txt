206. Reverse Linked list 

Given the head of a singly linked list, reverse the list, and return the reversed list.

Input: head = [1,2,3,4,5]
Output: [5,4,3,2,1]

class Solution {
    public ListNode reverseList(ListNode head) {

        ListNode temp=head;
        ListNode prev=null;
        ListNode front;
        while(temp != null){
            front=temp.next;
            temp.next=prev;
            prev=temp;
            temp=front;
            /*prev = head;
            head.next = front;
            temp.next = prev;*/

             
        }
        return prev;
    }
}
